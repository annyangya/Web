1. 简单示例：向div中动态添加文本内容
```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>DOM操作</title>

    <script>
        //动态添加
        function add() {
            var div=document.getElementById("div1")
            //创建节点p,label p
            var p=document.createElement("p")//<p></p>
            //添加文本
           var text=document.createTextNode("lllll")//p里面的内容

            //将p和文本内容关联
            p.appendChild(text)

            //将p添加到目标div中
            div.appendChild(p)
        }
    </script>

</head>
<body>

<input type="button" value="add" onclick="add()">

<!--动态向div中添加东西-->
<div id="div1">

</div>
</body>
</html>
```

#### 省市联动应用
在select中选中一个省，在另一个城市的select中显示相应省的城市
```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>


    <script>
        /*
        准备数据
         */
        var province1=[["深圳市","广州市","东莞市"],["绵阳市","成都市","南充市"],["长沙市","岳阳市","株洲市"]];
        /*
        1。确定事件：onchange
        2。函数：selectProvince（）
        3。函数内容：得到当前操作元素
                   得到当前选中的省份
                   从数组中取出对应的城市信息
                   动态创建城市元素节点，添加到城市select中
         */

        function selectProvince() {

            var city=document.getElementById("city")
            //得到当前选中的省份
            var province=document.getElementById("province")
            //获取城市信息
            var value=province.value
            var cities=province1[value]
            city.options.length=0//清空select中的option，以免显示混乱，将其他省的信息显示
            for(var i=0;i<cities.length;i++){
                var citytext=cities[i]

                //动态创建城市元素节点，添加到城市select中<option></option>
                var option=document.createElement("option")
                var text=document.createTextNode(citytext)
                option.appendChild(text)
                city.appendChild(option)
            }
        }
    </script>
    
</head>
<body>
<!--☑选择省份-->
<select onchange="selectProvince()" id="province">
    <option value="-1">请选择：</option>
    <option value="0">广东省</option>
    <option value="1">四川省</option>
    <option value="2">湖南省</option>
</select>

<!--☑选择城市，动态根据省份显示
DOM操作：Document Object Model：管理我们的文档增删改查规则
-->
<select id="city">

</select>

</body>
</html>
```
